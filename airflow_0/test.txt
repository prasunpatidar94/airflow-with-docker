from datetime import datetime, timedelta
from airflow.decorators import dag, task

default_args = {
    "owner": "Prasun Patidar",
    "retries": 5,
    "retry_delay": timedelta(minutes=5)
}

@dag(
    dag_id='Prasun_with_task_flow_api_airflow',
    default_args=default_args,
    description="Our first Python Operator DAG flow",
    start_date=datetime(2024, 5, 4),
    schedule_interval='@daily'
)
def hello_world_etl():

    @task()
    def get_name():
        return 'Prasun'
    
    @task()
    def get_age():
        return 27
    
    @task()
    def greet(name, age):
        print(f"Hi Prasun, my name is: {name} and age is: {age}")
    
    # Call the tasks inside the DAG function
    name = get_name()
    age = get_age()
    greet(name=name, age=age)

greet_dag = hello_world_etl()
